public class DateExtractor {

        if (matcher.find()) {
            System.out.println("Found date: " + matcher.group());

            System.out.println("Day: " + matcher.group(2));
            System.out.println("Year: " + matcher.group(3));
        } else {
            System.out.println("No date found in: \"" + text + "\"");
        }
    }
